print(f"샘플:\n{dataset[0]}")
# 출력된 컬럼명을 확인하고, 아래 formatting_func의 필드명을 맞춰 수정할 것

### 4.2 데이터 가공 (QA 형식 변환)

> ⚠️ `korean_textbooks`는 원본 그대로 쓰기보다 **태스크에 맞게 가공 후 사용**을 권장합니다.

```python
# 학습용 프롬프트 템플릿 (Gemma 4 Chat 형식)
def formatting_func(example):
    # ⚠️ Step 4.1에서 확인한 컬럼명에 따라 분기 처리
    # claude_evol 서브셋은 여러 형식이 가능:
    #   형식A: {"text": "..."}
    #   형식B: {"conversations": [{"from":"human","value":"..."}, {"from":"gpt","value":"..."}]}
    #   형식C: {"instruction": "...", "output": "..."}

    # 형식B: conversations 리스트 형태
    if "conversations" in example and example["conversations"]:
        convs = example["conversations"]
        user_text = ""
        assistant_text = ""
        for turn in convs:
            role = turn.get("from", "")
            value = turn.get("value", "")
            if role in ["human", "user"]:
                user_text = value[:1500]
            elif role in ["gpt", "assistant"]:
                assistant_text = value[:1500]
        if not user_text and not assistant_text:
            return {"formatted_text": ""}
        prompt = f"""<start_of_turn>user
{user_text}<end_of_turn>
<start_of_turn>model
{assistant_text}<end_of_turn>"""
        return {"formatted_text": prompt}

    # 형식C: instruction/output
    if "instruction" in example:
        instruction = example.get("instruction", "")
        output = example.get("output", "")
        if not instruction:
            return {"formatted_text": ""}
        prompt = f"""<start_of_turn>user
{instruction[:1500]}<end_of_turn>
<start_of_turn>model
{output[:1500]}<end_of_turn>"""
        return {"formatted_text": prompt}

    # 형식A: 단일 text 필드
    text = example.get("text", "")
    if not text:
        return {"formatted_text": ""}
    prompt = f"""<start_of_turn>user
다음 내용을 학습하고 핵심을 정리하세요:
{text[:1500]}<end_of_turn>
<start_of_turn>model
{text[:500]}<end_of_turn>"""
    return {"formatted_text": prompt}

# ⚡ 테스트용: 200건만 샘플링 (빠른 검증 목적)
# 실전 파인튜닝 시 → range(5000) 이상으로 변경
train_dataset = dataset.shuffle(seed=42).select(range(200))
train_dataset = train_dataset.map(formatting_func)

# 빈 데이터 필터링
train_dataset = train_dataset.filter(lambda x: len(x["formatted_text"]) > 0)

print(f"학습 데이터: {len(train_dataset)}건")
print(f"샘플 프롬프트:\n{train_dataset[0]['formatted_text'][:300]}...")
```

### 4.3 데이터 크기 가이드

| 데이터 규모 | Colab 학습 시간 | 용도 |
|-------------|----------------|------|
| **200건 (현재)** | **~5~10분** ⚡ | **최소 테스트 (현재 설정)** |

## Step 5 — 학습 실행

```python
from trl import SFTTrainer, SFTConfig

# ⚡ 테스트 최소 설정 — 목표: 5~10분 내 완료
# 실전 전환 시 아래 주석 참고하여 값 변경
trainer = SFTTrainer(
    model=model,
    tokenizer=tokenizer,
    train_dataset=train_dataset,
    args=SFTConfig(
        output_dir="./gemma4-e2b-korean-test",

        # 학습 설정 (테스트용 최소화)
        num_train_epochs=1,          # 테스트: 1 | 실전: 3
        max_steps=50,                # 50 steps 후 강제 종료 (약 3~5분)
                                     # 실전 전환 시 이 줄 삭제
        per_device_train_batch_size=1,  # 테스트: 1 | 실전: 2
        gradient_accumulation_steps=2,  # 실효 배치 = 1×2 = 2

        # 옵티마이저
        learning_rate=2e-4,
        weight_decay=0.01,
        warmup_steps=5,              # 테스트: 5 | 실전: 10

        # 정밀도
        fp16=not torch.cuda.is_bf16_supported(),
        bf16=torch.cuda.is_bf16_supported(),

        # 로깅 (테스트: 자주 출력)
        logging_steps=5,             # 테스트: 5 | 실전: 10
        save_strategy="no",          # 테스트: 저장 안함 | 실전: "epoch"

        # 성능
        optim="adamw_8bit",
        seed=42,

        # SFTConfig 전용 설정 (SFTTrainer에 직접 넣으면 에러 발생)
        max_seq_length=512,          # 테스트: 512 | 실전: 2048
        dataset_text_field="formatted_text",
        packing=False,               # 테스트: False | 실전: True
    ),
)
```

> **⚡ 테스트 → 실전 전환 체크리스트**
> - `max_steps=50` 줄 삭제
> - `num_train_epochs`: 1 → 3
> - `per_device_train_batch_size`: 1 → 2
> - `gradient_accumulation_steps`: 2 → 4
> - `save_strategy`: `"no"` → `"epoch"`
> - `max_seq_length`: 512 → 2048
> - `packing`: `False` → `True`
> - 데이터 샘플: 200 → 5,000 이상

## Step 6 — 학습 결과 테스트

> ⚠️ **Gemma 4는 멀티모달 모델**이라 `apply_chat_template`의 `content`를 문자열로 넘기면
> `TypeError: string indices must be integers` 에러 발생.
> **content를 `[{"type": "text", "text": "..."}]` 리스트 형식으로 전달해야 함.**

```python
# 추론 모드 전환
FastLanguageModel.for_inference(model)

# ✅ 방법 1: content를 리스트 형식으로 전달 (Gemma 4 멀티모달 형식)
messages = [
    {
        "role": "user",
        "content": [{"type": "text", "text": "인공지능의 핵심 개념 3가지를 설명하세요."}],
    },
]

inputs = tokenizer.apply_chat_template(
    messages,
    tokenize=True,
    add_generation_prompt=True,
    return_tensors="pt",
).to("cuda")

outputs = model.generate(
    input_ids=inputs,
    max_new_tokens=256,      # 테스트: 256 | 실전: 512
    temperature=0.7,
    top_p=0.9,
    do_sample=True,
)

response = tokenizer.decode(outputs[0], skip_special_tokens=True)
print(response)
```

```python
# ✅ 방법 2: tokenizer 직접 사용 (에러 없이 안전한 대안)

```bash

# 베이스 모델 (GGUF 파일 경로 — Step 8에서 확인한 실제 파일명 사용)
FROM ./unsloth.Q4_K_M.gguf

# 모델 파라미터
PARAMETER temperature 0.7
PARAMETER top_p 0.9
PARAMETER top_k 40
PARAMETER num_predict 1024
PARAMETER stop "<end_of_turn>"

# 시스템 프롬프트
SYSTEM """당신은 한국어에 특화된 AI 어시스턴트입니다.
정확하고 자연스러운 한국어로 응답하세요."""

# Gemma 공식 Chat 템플릿 (Go template 형식)
TEMPLATE """{{- range $i, $_ := .Messages }}
{{- $last := eq (len (slice $.Messages $i)) 1 }}
{{- if or (eq .Role "user") (eq .Role "system") }}<start_of_turn>user
{{ .Content }}<end_of_turn>
{{ if $last }}<start_of_turn>model
{{ end }}
{{- else if eq .Role "assistant" }}<start_of_turn>model
{{ .Content }}{{ if not $last }}<end_of_turn>
{{ end }}
{{- end }}
{{- end }}"""
EOF
```

## 트러블슈팅

| 증상 | 원인 | 해결 |
|------|------|------|
| Colab `CUDA OOM` | VRAM 부족 | `batch_size=1`, `r=8`, `max_seq_length=1024` |
| Loss가 안 줄어듦 | 학습률 문제 | `learning_rate`를 `1e-4`~`5e-4` 범위 조절 |
| GGUF 변환 실패 | 디스크 부족 | Colab 디스크 정리 후 재시도 |
| Ollama 모델 로드 실패 | GGUF 경로 오류 | Modelfile의 `FROM` 경로를 절대경로로 변경 |
| 한국어 응답 품질 낮음 | 데이터 부족/품질 | 데이터를 10,000건+로 증가, 에포크 조절 |
| 맥북 응답 느림 | 모델 크기 | `q4_k_m` 양자화 사용, `num_predict` 줄이기 |
| Colab 세션 끊김 | 시간 제한 | `save_strategy="steps"`, Google Drive 마운트 |
